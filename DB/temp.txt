CREATE  TABLE Course(CNO INT ,
                    CNAME CHAR(20),
                    TNO INT ,
                    CREDIT INT,
                    PRIMARY KEY (CNO,TNO)
                    );


CREATE TABLE SC(SNO INT ,
                CNO INT,
                GRADE INT,
                PRIMARY KEY(CNO,SNO));

DROP TABLE IF EXISTS Teacher;
CREATE  TABLE Teacher(TNO int  ,
                        TNAME CHAR(8) NOT NULL,
                        DEPTNO INT,
                        PRIMARY KEY(TNO));

CREATE TABLE Dept(DEPTNO INT,
                    DNAME CHAR(20),
                    PRIMARY KEY(DEPTNO));
